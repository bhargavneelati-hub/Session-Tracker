Connect to Google Sheets

Set up once. After this every device that opens your URL shares the same sessions automatically.

  1. Create a Google Sheet. Name it Session Log
  2. Add three tabs named exactly: Session Log   Sessions   Config
  3. In Session Log row 1 add headers:
    Date | Task | Domain | Novelty | Stakes | Complexity | Prediction | Confidence | Loop | Clarity | Anxiety | Phrase | Outcome | Quality | Time | Expected_Time | Pace | Reflection
  4. In Sessions row 1 add headers:
    id | date | status | task | domain | novelty | stakes | complexity | expected_time | prediction | confidence | loop | clarity | anxiety | phrase | outcome | outcome_note | quality | time | pace_ok | intent_match | intent_note | reflection | type
  5. In Config: cell A1 = script_url, leave B1 empty
  6. Go to Extensions → Apps Script, delete all code, paste the script below, save
  7. Deploy → New deployment → Web app → Execute as: Me → Who has access: Anyone → Deploy
  8. Copy the Web App URL (ends in /exec) and paste it below:
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var action = e.parameter.action || 'log';
  var cb = e.parameter.callback || '';

  if (action === 'ping') {
    return json({status:'ok'}, cb);
  }
  if (action === 'setUrl') {
    var cfg = ss.getSheetByName('Config');
    if (cfg) cfg.getRange('B1').setValue(e.parameter.url);
    return json({status:'ok'}, cb);
  }
  if (action === 'getSessions') {
    var sh = ss.getSheetByName('Sessions');
    if (!sh || sh.getLastRow() < 2) return json({sessions:[]}, cb);
    var data = sh.getRange(2,1,sh.getLastRow()-1,sh.getLastColumn()).getValues();
    var hdrs = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
    var sessions = data.map(function(row){
      var o={};hdrs.forEach(function(h,i){o[h]=row[i];});return o;
    });
    return json({sessions:sessions}, cb);
  }
  if (action === 'upsertSession') {
    var sh = ss.getSheetByName('Sessions');
    var hdrs = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
    var p = e.parameter; var id = p.id; var found = -1;
    if (sh.getLastRow() > 1) {
      var ids = sh.getRange(2,1,sh.getLastRow()-1,1).getValues();
      for (var i=0;i 0){sh.getRange(found,1,1,row.length).setValues([row]);}
    else{sh.appendRow(row);}
    return json({status:'ok'}, cb);
  }
  if (action === 'completeSession') {
    var log = ss.getSheetByName('Session Log') || ss.getActiveSheet();
    var p = e.parameter;
    log.appendRow([
      p.date, p.task, p.domain, p.novelty, p.stakes,
      p.complexity, p.prediction, p.confidence, p.loop,
      p.clarity||'not-recorded', p.anxiety||'', p.phrase||'not-recorded',
      p.outcome, p.quality, p.time, p.expected_time,
      p.pace_ok||'', p.reflection
    ]);
    var sh = ss.getSheetByName('Sessions');
    if (sh && sh.getLastRow() > 1) {
      var ids = sh.getRange(2,1,sh.getLastRow()-1,1).getValues();
      for (var i=ids.length-1;i>=0;i--){
        if(ids[i][0]==p.id){sh.deleteRow(i+2);break;}
      }
    }
    return json({status:'ok'}, cb);
  }
  return json({error:'unknown action'}, cb);
}
function json(obj, cb) {
  if (cb) {
    return ContentService
      .createTextOutput(cb+'('+JSON.stringify(obj)+')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
