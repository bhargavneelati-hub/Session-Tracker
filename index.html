Connect to Google Sheets

This takes about 10 minutes once. After setup, every device that opens this URL shares the same sessions — no repeat configuration needed.

  1. Create a new Google Sheet. Name it Session Log
  2. Add three tabs (sheets) named exactly: Session Log  Sessions  Config
  3. In Session Log row 1 paste:
    Date | Task | Domain | Novelty | Stakes | Complexity | Prediction | Confidence | Loop | Clarity | Anxiety | Phrase | Outcome | Quality | Time | Expected Time | Pace OK | Intent Match | Reflection
  4. In Sessions row 1 paste:
    id | date | status | task | domain | novelty | stakes | complexity | expected_time | prediction | confidence | loop | clarity | anxiety | phrase | outcome | outcome_note | quality | time | pace_ok | intent_match | intent_note | reflection | type
  5. In Config cell A1 type: script_url   leave B1 empty for now
  6. Go to Extensions → Apps Script, delete all existing code, paste the script below, save
  7. Click Deploy → New deployment → Web app → Execute as: Me → Who has access: Anyone → Deploy
  8. Copy the Web App URL and paste it here:
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var action = e.parameter.action || 'log';

  if (action === 'ping') {
    return json({status:'ok'});
  }

  if (action === 'getUrl') {
    var cfg = ss.getSheetByName('Config');
    var url = cfg ? cfg.getRange('B1').getValue() : '';
    return json({url: url});
  }

  if (action === 'setUrl') {
    var cfg = ss.getSheetByName('Config');
    if (cfg) cfg.getRange('B1').setValue(e.parameter.url);
    return json({status:'ok'});
  }

  if (action === 'getSessions') {
    var sh = ss.getSheetByName('Sessions');
    if (!sh || sh.getLastRow() < 2) return json({sessions:[]});
    var data = sh.getRange(2, 1, sh.getLastRow()-1, sh.getLastColumn()).getValues();
    var headers = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
    var sessions = data.map(function(row) {
      var obj = {};
      headers.forEach(function(h,i){ obj[h] = row[i]; });
      return obj;
    });
    return json({sessions: sessions});
  }

  if (action === 'upsertSession') {
    var sh = ss.getSheetByName('Sessions');
    var headers = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
    var p = e.parameter;
    var id = p.id;
    // find existing row
    var found = -1;
    if (sh.getLastRow() > 1) {
      var ids = sh.getRange(2,1,sh.getLastRow()-1,1).getValues();
      for (var i=0; i 0) {
      sh.getRange(found, 1, 1, row.length).setValues([row]);
    } else {
      sh.appendRow(row);
    }
    return json({status:'ok'});
  }

  if (action === 'completeSession') {
    // write to Session Log (permanent record)
    var log = ss.getSheetByName('Session Log') || ss.getActiveSheet();
    var p = e.parameter;
    log.appendRow([
      p.date, p.task, p.domain, p.novelty, p.stakes,
      p.complexity, p.prediction, p.confidence, p.loop,
      p.clarity, p.anxiety, p.phrase, p.outcome,
      p.quality, p.time, p.expected_time, p.pace_ok,
      p.intent_match, p.reflection
    ]);
    // remove from Sessions tab
    var sh = ss.getSheetByName('Sessions');
    if (sh && sh.getLastRow() > 1) {
      var ids = sh.getRange(2,1,sh.getLastRow()-1,1).getValues();
      for (var i=ids.length-1; i>=0; i--) {
        if (ids[i][0] == p.id) { sh.deleteRow(i+2); break; }
      }
    }
    return json({status:'ok'});
  }

  return json({error:'unknown action'});
}

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
