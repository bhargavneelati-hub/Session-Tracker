function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Session Log") || ss.getActiveSheet();
  var p = e.parameter;
  sheet.appendRow([
    p.date, p.task, p.domain, p.novelty, p.stakes,
    p.complexity, p.prediction, p.confidence, p.loop,
    p.clarity, p.anxiety, p.phrase, p.outcome,
    p.quality, p.time, p.expected_time, p.pace_ok, p.reflection
  ]);
  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
